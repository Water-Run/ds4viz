# `ds4viz`: C编码规范

在编写C代码时，你需要遵循以下编码规范：

**语言级别:** `C17`  
**类型注解:** C是静态强类型语言，所有变量必须明确声明类型。使用`const`、`restrict`、`volatile`等类型限定符。善用`stdint.h`中的固定宽度整数类型(`int32_t`、`uint64_t`等)、`stdbool.h`中的`bool`类型  
**编码规范:** 遵循经典K&R和GNU风格的混合体。使用Allman风格(大括号独占一行)。代码需要在GCC/Clang最严格模式(`-Wall -Wextra -Wpedantic -Werror`)下无Warning。采用传统C写法，包括：所有变量在块开头声明、使用`typedef`定义类型别名、使用函数指针、使用结构体和联合体、避免使用VLA(可变长度数组)、使用`goto`进行错误清理(仅限必要时)  
**内存管理:** 严格遵循"谁分配谁释放"原则，所有`malloc`/`calloc`必须有对应的`free`，避免内存泄漏和悬空指针  
**文档注释:** 在文件开头、所有函数、结构体、枚举、宏定义上都要有遵循Doxygen风格的文档注释。代码即文档的编码规范：代码需要尽可能的自描述、易懂，减少非文档注释之外的注释内容。

## 命名约定

- **小写+下划线**: 函数名、变量名 (如 `get_user_name`, `total_count`)
- **大写+下划线**: 宏定义、枚举常量、全局常量 (如 `MAX_SIZE`, `DEFAULT_TIMEOUT`)
- **小写+下划线+_t后缀**: typedef类型别名 (如 `data_processor_t`, `status_t`)
- **小写+下划线+_s后缀**: 结构体标签 (如 `struct data_processor_s`)
- **小写+下划线+_e后缀**: 枚举标签 (如 `enum status_e`)
- **小写+下划线+_u后缀**: 联合体标签 (如 `union value_u`)
- **匈牙利命名法(可选)**: 指针使用`p`前缀 (如 `p_data`)

## 文件结构

每个C源文件必须按以下顺序组织：

1. 文件头注释
2. 包含指令(`#include`)
3. 宏定义
4. 类型定义(`typedef`, `struct`, `enum`, `union`)
5. 静态全局变量声明
6. 静态函数声明
7. 外部函数实现
8. 静态函数实现

每个C头文件必须包含：

1. 文件头注释
2. 头文件保护符(`#ifndef`)
3. 包含指令
4. 宏定义
5. 类型定义
6. 函数声明
7. 头文件保护符结束(`#endif`)

### 源文件 (data_processor.c)

```c
/**
 * @file data_processor.c
 * @brief 数据处理器实现
 * 
 * 实现数据处理器的核心功能，包括数据转换、批处理等
 * 
 * @author WaterRun
 * @date 2025-12-23
 * @version 1.0.0
 */

#include "data_processor.h"
#include <stdlib.h>
#include <string.h>
#include <stdio.h>
#include <assert.h>

/* ==================== 内部宏定义 ==================== */

/**
 * @brief 检查指针是否为NULL
 * 
 * @param ptr 要检查的指针
 * @return 是否为NULL
 */
#define IS_NULL(ptr) ((ptr) == NULL)

/**
 * @brief 最小值宏
 * 
 * @param a 第一个值
 * @param b 第二个值
 * @return 两者中的较小值
 */
#define MIN(a, b) ((a) < (b) ? (a) : (b))

/* ==================== 静态函数声明 ==================== */

/**
 * @brief 验证字符串是否为空
 * 
 * @param str 要验证的字符串
 * @return 如果为空或只包含空白字符返回true，否则返回false
 */
static bool is_empty_string(const char *str);

/**
 * @brief 分配并复制字符串
 * 
 * @param src 源字符串
 * @return 新分配的字符串副本，失败返回NULL
 * 
 * @note 调用者负责释放返回的内存
 */
static char *string_duplicate(const char *src);

/* ==================== 外部函数实现 ==================== */

data_processor_t *data_processor_create(const char *connection_string, size_t batch_size)
{
    data_processor_t *processor = NULL;
    
    /* 参数验证 */
    if (IS_NULL(connection_string) || is_empty_string(connection_string))
    {
        fprintf(stderr, "错误: connection_string不能为空\n");
        return NULL;
    }
    
    if (batch_size == 0)
    {
        fprintf(stderr, "错误: batch_size必须大于0\n");
        return NULL;
    }
    
    /* 分配处理器内存 */
    processor = (data_processor_t *)calloc(1, sizeof(data_processor_t));
    if (IS_NULL(processor))
    {
        fprintf(stderr, "错误: 内存分配失败\n");
        return NULL;
    }
    
    /* 复制连接字符串 */
    processor->connection_string = string_duplicate(connection_string);
    if (IS_NULL(processor->connection_string))
    {
        free(processor);
        fprintf(stderr, "错误: 字符串复制失败\n");
        return NULL;
    }
    
    /* 初始化字段 */
    processor->batch_size = batch_size;
    processor->status = STATUS_PENDING;
    processor->success_count = 0;
    processor->failure_count = 0;
    processor->private_data = NULL;
    
    return processor;
}

void data_processor_destroy(data_processor_t *processor)
{
    if (IS_NULL(processor))
    {
        return;
    }
    
    /* 释放连接字符串 */
    SAFE_FREE(processor->connection_string);
    
    /* 释放私有数据(如果有的话) */
    SAFE_FREE(processor->private_data);
    
    /* 释放处理器本身 */
    free(processor);
}

int data_processor_process(
    data_processor_t *processor,
    const void *data,
    size_t size,
    transformer_fn_t transformer,
    void *context,
    process_result_t *result)
{
    void *output_buffer = NULL;
    size_t output_buffer_size = 0;
    int transform_result = 0;
    
    /* 参数验证 */
    if (IS_NULL(processor) || IS_NULL(data) || IS_NULL(transformer) || IS_NULL(result))
    {
        fprintf(stderr, "错误: 参数不能为NULL\n");
        return ERROR;
    }
    
    if (size == 0)
    {
        fprintf(stderr, "错误: size必须大于0\n");
        return ERROR;
    }
    
    /* 更新状态 */
    processor->status = STATUS_PROCESSING;
    
    /* 分配输出缓冲区(假设输出大小与输入相同) */
    output_buffer_size = size * 2; /* 预留额外空间 */
    output_buffer = malloc(output_buffer_size);
    if (IS_NULL(output_buffer))
    {
        fprintf(stderr, "错误: 输出缓冲区分配失败\n");
        processor->status = STATUS_FAILED;
        return ERROR;
    }
    
    /* 执行转换 */
    transform_result = transformer(data, size, output_buffer, output_buffer_size, context);
    
    if (transform_result < 0)
    {
        /* 转换失败 */
        free(output_buffer);
        processor->status = STATUS_FAILED;
        processor->failure_count++;
        
        result->status = STATUS_FAILED;
        result->success_count = 0;
        result->failure_count = 1;
        result->output_data = NULL;
        result->output_size = 0;
        
        return ERROR;
    }
    
    /* 转换成功 */
    processor->status = STATUS_COMPLETED;
    processor->success_count++;
    
    result->status = STATUS_COMPLETED;
    result->success_count = 1;
    result->failure_count = 0;
    result->output_data = output_buffer;
    result->output_size = (size_t)transform_result;
    
    return SUCCESS;
}

int data_processor_batch_process(
    data_processor_t *processor,
    const void **data_array,
    size_t item_count,
    size_t item_size,
    transformer_fn_t transformer,
    void *context,
    process_result_t *result)
{
    size_t i = 0;
    size_t batch_start = 0;
    size_t batch_end = 0;
    size_t success_total = 0;
    size_t failure_total = 0;
    void *combined_output = NULL;
    size_t combined_size = 0;
    
    /* 参数验证 */
    if (IS_NULL(processor) || IS_NULL(data_array) || IS_NULL(transformer) || IS_NULL(result))
    {
        fprintf(stderr, "错误: 参数不能为NULL\n");
        return ERROR;
    }
    
    if (item_count == 0 || item_size == 0)
    {
        fprintf(stderr, "错误: item_count和item_size必须大于0\n");
        return ERROR;
    }
    
    processor->status = STATUS_PROCESSING;
    
    /* 批量处理 */
    for (batch_start = 0; batch_start < item_count; batch_start += processor->batch_size)
    {
        batch_end = MIN(batch_start + processor->batch_size, item_count);
        
        /* 处理当前批次 */
        for (i = batch_start; i < batch_end; i++)
        {
            process_result_t item_result = {0};
            int process_status = 0;
            
            if (IS_NULL(data_array[i]))
            {
                failure_total++;
                continue;
            }
            
            process_status = data_processor_process(
                processor,
                data_array[i],
                item_size,
                transformer,
                context,
                &item_result
            );
            
            if (process_status == SUCCESS)
            {
                success_total += item_result.success_count;
                /* TODO: 合并输出数据 */
                SAFE_FREE(item_result.output_data);
            }
            else
            {
                failure_total += item_result.failure_count;
            }
        }
    }
    
    /* 设置最终状态 */
    processor->status = (failure_total == 0) ? STATUS_COMPLETED : STATUS_FAILED;
    
    result->status = processor->status;
    result->success_count = success_total;
    result->failure_count = failure_total;
    result->output_data = combined_output;
    result->output_size = combined_size;
    
    return SUCCESS;
}

{

static bool is_empty_string(const char *str)
{
    if (IS_NULL(str))
    {
        return true;
    }
    
    while (*str != '\0')
    {
        if (*str != ' ' && *str != '\t' && *str != '\n' && *str != '\r')
        {
            return false;
        }
        str++;
    }
    
    return true;
}

static char *string_duplicate(const char *src)
{
    char *dst = NULL;
    size_t len = 0;
    
    if (IS_NULL(src))
    {
        return NULL;
    }
    
    len = strlen(src);
    dst = (char *)malloc(len + 1);
    
    if (IS_NULL(dst))
    {
        return NULL;
    }
    
    memcpy(dst, src, len);
    dst[len] = '\0';
    
    return dst;
}
```

## 最佳实践

1. **Allman风格**: 大括号独占一行
2. **变量声明**: 所有变量在块开头声明
3. **内存管理**: 严格配对malloc/free，使用SAFE_FREE宏
4. **错误处理**: 使用返回值表示成功/失败，使用goto清理资源(必要时)
5. **const正确性**: 尽可能使用const修饰不变参数
6. **头文件保护**: 使用#ifndef/#define/#endif保护
7. **extern "C"**: 头文件支持C++编译器
8. **断言**: 使用assert()验证内部逻辑(仅用于调试)
9. **固定宽度类型**: 使用stdint.h中的类型
10. **文档完整性**: 所有公开接口都必须有完整的Doxygen注释
